## 3. getline

### 3.1. 範例：一次讀一行

## 12. C 語言：字串函數

### 12.1. C 語言：範例：strlen

計算字串長度(不含`\0`)

```c
int strlen(char *string); // 接受一個字元指標，回傳長度(整數)
```

{% raw %}

```c
#include <stdio.h>
#include <string.h>
#define STRINGLEN 80
int main(void){
    char string[STRINGLEN];
    scanf("%s", string);
    printf("%s\n", string);

    int length = strlen(string); // 取出長度

    // 反轉
    for (int i = 0; i < length / 2; i++) {
        char temp = string[i];
        string[i] = string[length -i - 1];
        string[length - i - 1] = temp;
    }
    printf("%s\n", string);
    return 0;
}
```

### 12.7. C 語言：範例：字串查找：搜尋目標字串是否存在於陣列中

{% raw %}

```c
#include <stdio.h>
#include <string.h>

int main() {
    char *fruits[] = {"apple", "banana", "grape", "melon", "kiwi"};
    int size = sizeof(fruits) / sizeof(fruits[0]);
    char target[20];

    printf("請輸入你要找的水果名稱：");
    scanf("%s", target);

    int found = 0;
    for (int i = 0; i < size; i++) {
        if (strcmp(fruits[i], target) == 0) {
            found = 1;
            printf("找到 %s 在第 %d 個位置。\n", target, i);
            break;
        }
    }

    if (!found) {
        printf("找不到 %s。\n", target);
    }

    return 0;
}
```

### 12.8. C 語言：範例：字串排序：輸入多個字串，將它們依照字典順序排序

{% raw %}

```c
#include <stdio.h>
#include <string.h>

#define SIZE 5
#define LEN 20

int main() {
    char names[SIZE][LEN] = {
        "banana",
        "apple",
        "grape",
        "kiwi",
        "melon"
    };

    // Bubble Sort
    for (int i = 0; i < SIZE - 1; i++) {
        for (int j = 0; j < SIZE - 1 - i; j++) {
            if (strcmp(names[j], names[j + 1]) > 0) {
                char temp[LEN];
                strcpy(temp, names[j]);
                strcpy(names[j], names[j + 1]);
                strcpy(names[j + 1], temp);
            }
        }
    }

    printf("排序後的字串：\n");
    for (int i = 0; i < SIZE; i++) {
        printf("%s\n", names[i]);
    }

    return 0;
}
```

### 12.9. C 語言：strtok

```c
char *strtok(char *str, const char *delim);
// str：要被切割的字串，第一次傳入要給原始字串
// delim：分隔符（delimiters），例如：空格 " "、逗號 ","、冒號 ":" 等。
```

回傳值：指向切割後的 第一個字串區塊 的指標，找不到時回傳 NULL。

之後繼續呼叫 strtok(NULL, delim)，會接續從上一次中斷的位置繼續切割。

{% raw %}

```c
#include <stdio.h>
#include <string.h>

int main() {
    char str[] = "apple,banana,grape,melon,kiwi";
    char *token;

    // 第一次呼叫：傳入原始字串
    token = strtok(str, ",");

    // 持續呼叫：傳入 NULL
    while (token != NULL) {
        printf("%s\n", token);
        token = strtok(NULL, ",");  // 繼續切
    }
    return 0;
}
```
