## 基本用法

-   一名學生有以下資料：
    -   name：string
    -   student ID：string
    -   phone：string
    -   各學期平均成績：float[]
    -   birth：3 個 int

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <stdint.h>

struct student {
    char name[10];
    char id[20];
    char phone[20];
    float grade[4]; // 16
    int birthYear, birthMonth, birthDay; // 12
};

int main(void){
    struct student john = {
        "John Smith",
        "12345",
        "1234567",
        {4.0, 3.9, 3.8, 3.6},
        2000, 1, 1
    };
    printf("name is %s\n", john.name);
    printf("id is %s\n", john.id);
    printf("phone is %s\n", john.phone);
    printf("john.grade[0] is %f\n", john.grade[0]);
    return 0;
}

// name is John Smith12345
// id is 12345
// phone is 1234567
// john.grade[0] is 4.000000
```
